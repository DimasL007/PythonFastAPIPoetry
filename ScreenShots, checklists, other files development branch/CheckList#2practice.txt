wsl --update — оновлення підсистеми Windows для коректної роботи Docker.

docker compose up -d --build — збірка образів та запуск контейнерів у фоновому режимі.

docker ps — перевірка статусу та отримання ID запущених контейнерів.

docker exec -it container_id poetry show — вивід списку бібліотек всередині робочого контейнера.
